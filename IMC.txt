# Solicitando ao usuário que informe seu peso e altura
peso = float(input("Digite o seu peso em kg: "))
altura = float(input("Digite a sua altura em metros: "))

# Calculando o IMC e armazenando o resultado em uma variável
imc = peso / altura ** 2

# Imprimindo o resultado do IMC na tela
print("Seu IMC é:", imc)
